  // Schema FAQPage
  generateFAQPage(faqs) {
    return {
      "@context": "https://schema.org",
      "@type": "FAQPage",
      "mainEntity": faqs.map(faq => ({
        "@type": "Question",
        "name": faq.question,
        "acceptedAnswer": {
          "@type": "Answer",
          "text": faq.answer
        }
      }))
    };
  }

  // Schema BreadcrumbList
  generateBreadcrumb(items) {
    return {
      "@context": "https://schema.org",
      "@type": "BreadcrumbList",
      "itemListElement": items.map((item, index) => ({
        "@type": "ListItem",
        "position": index + 1,
        "name": item.name,
        "item": item.url
      }))
    };
  }

  // Injetar schema na p√°gina
  injectSchema(schemaObject) {
    const script = document.createElement('script');
    script.type = 'application/ld+json';
    script.text = JSON.stringify(schemaObject);
    document.head.appendChild(script);
  }

  // Gerar todos os schemas da p√°gina
  generateAll() {
    // LocalBusiness em todas as p√°ginas
    this.injectSchema(this.generateLocalBusiness());

    // Schemas espec√≠ficos por tipo de p√°gina
    if (this.pageType === 'bairro' || this.pageType === 'cidade') {
      // Breadcrumb
      const breadcrumb = this.generateBreadcrumb([
        { name: "Home", url: "https://www.servicosadp.shop/" },
        { name: this.pageType === 'bairro' ? 'Bairros' : 'Cidades', url: `https://www.servicosadp.shop/${this.pageType}s/` },
        { name: this.pageData.name, url: this.pageData.url }
      ]);
      this.injectSchema(breadcrumb);

      // FAQ (se existir)
      if (this.pageData.faqs && this.pageData.faqs.length > 0) {
        this.injectSchema(this.generateFAQPage(this.pageData.faqs));
      }
    }
  }
}
